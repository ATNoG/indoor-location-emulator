 85 /**
 86  * Load a JSON object from the path
 87  * @param {string} url URL string path
 88  * @param {XMLHttpRequest} callback Callback function
 89  */
 90 function loadJsonFile(url, callback) {
 91     var xobj = new XMLHttpRequest();
 92     xobj.overrideMimeType("application/json");
 93     xobj.open('GET', url, true); // Replace 'my_data' with the path to your file
 94     xobj.onreadystatechange = function () {
 95         if (xobj.readyState == 4 && xobj.status == "200") {
 96             // Required use of an anonymous callback as .open will NOT return a value but simply returns undefined in asynchronous mode
 97             callback(xobj.responseText);
 98         }
 99     };
100     xobj.send(null);
101 }

151 /**
152  * Generate UUID string
153  * @return {string} Generated 16-characters UUID string
154  */  
155 function generateUUID() {
156     var d = new Date().getTime(); //Timestamp
157     var d2 = ((typeof performance !== 'undefined') && performance.now && (performance.now() * 1000)) || 0; //Time in microseconds since page-load or 0 if unsupported
158     return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function (c) {
159         var r = Math.random() * 16; //random number between 0 and 16
160         if (d > 0) { //Use timestamp until depleted
161             r = (d + r) % 16 | 0;
162             d = Math.floor(d / 16);
163         } else { //Use microseconds since page-load if supported
164             r = (d2 + r) % 16 | 0;
165             d2 = Math.floor(d2 / 16);
166         }
167         return (c === 'x' ? r : (r & 0x3 | 0x8)).toString(16);
168     });
169 }

181 /**
182  * Get bearing between two point locations
183  * @typedef {Object} LongLat with Longitude and Latitude coordinates
184  * @property {number} Long - Longitude value
185  * @property {number} Lat - Latitude value
186  * @param {LongLat} LongLat1 - Longitude and Latitude coordinates of point 1
187  * @param {LongLat} LongLat2 - Longitude and Latitude coordinates of point 2
188  * @return {number} Bearing between the both point locations
189  */  
190 function getBearingBetweenLocations(LongLat1, LongLat2) {
191     // convert Long and Lat to Radians
192     var long1 = LongLat1[0] * Math.PI / 180;
193     var lat1 = LongLat1[1] * Math.PI / 180;
194     var long2 = LongLat2[0] * Math.PI / 180;
195     var lat2 = LongLat2[1] * Math.PI / 180;
196 
197     // calculate differences
198     var deltaLong = (long2 - long1);
199     var deltaLat = (lat2 - lat1);
200 
201     // get angle in randians 
202     var brng = Math.atan2(deltaLat, deltaLong);
203 
204     // convert to degrees, rotate -90deg and restrict angle from -180deg to 180deg
205     brng = rescrictTo180Degrees(brng * 180 / Math.PI - 90);
206 
207     return brng;
208 }

210 /**
211  * Remove an item occurrency from an array
212  * @param {Array} arr Array from where we want remove element
213  * @param {any} value Element to be removed
214  * @returns {Array} Argument array without removed element
215  */ 
216 function removeItemOnce(arr, value) {
217     var index = arr.indexOf(value);
218     if (index > -1) {
219         arr.splice(index, 1);
220     }
221     return arr;
222 }

224 /**
225  * Remove all item occurrencies from an array
226  * @param {Array} arr Array from where we want remove element
227  * @param {any} value Element to be removed
228  * @returns {Array} Argument array without all occurrencies of removed element
229  */ 
230 function removeItemAll(arr, value) {
231     var i = 0;
232     while (i < arr.length) {
233         if (arr[i] === value) {
234             arr.splice(i, 1);
235         } else {
236             ++i;
237         }
238     }
239     return arr;
240 }

242 /**
243  * Rectrict the value of an angle to be between -180° to 180° (continuous mapping problem)
244  * @param {number} value Value where to apply the restriction
245  * @returns {number} Angle restricted between -180° to 180°
246  */ 
247 function rescrictTo180Degrees(value) {
248     // reduce the angle  
249     var angle = parseFloat(value) % 360;
250 
251     // force it to be the positive remainder, so that 0 <= angle < 360  
252     angle = (angle + 360) % 360;
253 
254     // force into the minimum absolute value residue class, so that -180 < angle <= 180  
255     if (angle > 180)
256         angle -= 360;
257 
258     return angle;
259 }
260 /**
261  * Conversion of Latitude and Longitude, from Decimal Degrees Format to DMS [Degrees° Minutes' Secconds"]
262  * @param {number} coordinate Value where to apply the conversion
263  * @returns {string} Coverted value to DMS [Degrees° Minutes' Secconds"]
264  */  
265 function toDegreesMinutesAndSeconds(coordinate) {
266     var absolute = Math.abs(coordinate);
267     var degrees = Math.floor(absolute);
268     var minutes = Math.floor((absolute - degrees) * 60);
269     var seconds = ((absolute - degrees - minutes / 60) * Math.pow(60, 2)).toFixed(2);
270     var deg_symbol = String.fromCharCode(176); // degree symbol (°)
271     return degrees + deg_symbol + " " + minutes + "' " + seconds + """;
272 }

274 /**
275  * Convert Longitude from Decimal Degree Format to DMS (Degrees° Minutes' Seconds")
276  * @param {number} lng Longitude value where to apply the conversion
277  * @returns {string} Coverted longitude value to DMS [Degrees° Minutes' Secconds"]
278  */ 
279 function convertDMSLng(lng) {
280     var longitude = toDegreesMinutesAndSeconds(lng);
281     var longitudeCardinal = lng >= 0 ? "E" : "W";
282     return longitude + " " + longitudeCardinal;
283 }

290 function convertDMSLat(lat) {
291     var latitude = toDegreesMinutesAndSeconds(lat);
292     var latitudeCardinal = lat >= 0 ? "N" : "S";
293     return latitude + " " + latitudeCardinal;
294 }

296 /**
297  * Definition of Geocoder Search Bar
298  * @param {string} query Query string to search
299  * @param {Array} mapFeaturesData Array of Map features
300  * @returns {Array} Array of queried matching features
301  */ 
302 function forwardGeocoder(query, mapFeaturesData) {
303     var matchingFeatures = [];
304 
305     for (var i = 0, len = mapFeaturesData.features.length; i < len; i++) {
306         var feature = mapFeaturesData.features[i];
307         if (feature.properties.hasOwnProperty('name') &&
308             feature.properties['name']
309                 .toLowerCase()
310                 .search(query.toLowerCase()) !== -1
311         ) {
312             feature['place_name'] = feature.properties['name'];
313             feature['center'] = turf.centroid(feature).geometry.coordinates;
314             feature['place_type'] = ['park'];
315             matchingFeatures.push(feature);
316         } else if (feature.properties.hasOwnProperty('name') &&
317             feature.properties.name
318                 .toLowerCase()
319                 .search(query.toLowerCase()) !== -1
320         ) {
321             feature['place_name'] = feature.properties.name;
322             feature['center'] = turf.centroid(feature).geometry.coordinates;
323             feature['place_type'] = ['park'];
324             matchingFeatures.push(feature);
325         }
326     }
327     return matchingFeatures;
328 }

330 /**
331  * Catch Walls Of Map Features Data array and filter
332  * @param {Array} mapFeaturesData Array of Map features
333  * @param {Array} geojson_walls Array of Map walls features
334  * @param {Array} filter_walls_keywords Array of keywords to filter Map walls features 
335  * @param {Array} filter_walls_level Array of levels to filter Map walls features 
336  * @returns {Array} Array of geojson_walls filtered by arguments keywords and levels
337  */ 
338 function catchWallsOfMapFeaturesData(mapFeaturesData, geojson_walls, filter_walls_keywords, filter_walls_level) {
339     // deal with obstacle walls
340     mapFeaturesData.features.forEach(index => {
341         // apply polygon to line to define walls
342         wall = turf.polygonToLine(index)
343         // apply filter data (for the given "words" on the given "levels")
344         filter_walls_level.forEach(level => {
345             filter_walls_keywords.forEach(word => {
346                 // apply filter data (ex policlínica: for "walls" and "glass", only on level 0)
347                 if (wall.properties.name.includes(word) && wall.properties.level.includes(level)) {
348                     geojson_walls.features.push(wall);
349                 }
350             })
351         })
352     });
353 }
