 65    def updateAssetPoint(self, apArray, uuid):
 66        """
 67        Update an Asset Point object on the GOD class object.
 68
 69        Args:
 70            self: GOD object itself
 71            apArray: Array of Asset Points objects
 72            uuid: Frontend session UUID
 73        """
 74        self.assetPoints[uuid] = {}
 75
 76        for assetPoint in range(len(apArray)):
 77            apCoords = apArray[assetPoint]['LongLat']
 78            largeDirOpen = apArray[assetPoint]['LargeDirOpen']
 79            smallDirOpen = apArray[assetPoint]['SmallDirOpen']
 80            if len(self.assetPoints[uuid]) > assetPoint:
 81                apObject = self.getAssetPoint(assetPoint, uuid)
 82                apObject.setCoords(apCoords[0], apCoords[1])
 83                apObject.setLargeDirectionOpening(
 84                    largeDirOpen[0], largeDirOpen[1])
 85                apObject.setSmallDirectionOpening(
 86                    smallDirOpen[0], smallDirOpen[1])
 87            else:
 88                newAp = AssetPoint(apCoords[0], apCoords[1], largeDirOpen[0],
 89                                   largeDirOpen[1], smallDirOpen[0], smallDirOpen[1])
 90                self.addAssetPoint(assetPoint, newAp, uuid)

 94        Update an Antenna object on the GOD class object.

 96        Args:

113            else:
114                newant = Antenna(
