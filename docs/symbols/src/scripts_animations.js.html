  5 /**
  6  * Activate Asset Point Animations
  7  * @param {FeatureCollection} geojson_asset_points Asset Points FeatureCollection
  8  * @param {FeatureCollection} geojson_assetpoints_directions Asset Points Directions FeatureCollection
  9  * @param {FeatureCollection} geojson_anchors Anchors FeatureCollection
 10  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection
 11  * @param {FeatureCollection} wall_intersections_points Map Features Walls Points Intersections FeatureCollection
 12  * @param {FeatureCollection} geojson_lines_of_sight Lines of Sight FeatureCollection
 13  * @param {FeatureCollection} geojson_walls Map Features Walls FeatureCollection
 14  * @param {Array} pulsing_dots_layers_sources Pulsing Dots sources Array
 15  * @param {Array} asset_point_animations Asset Point Animations Array
 16  * @param {number} cpu_frame_rate CPU frame rate
 17  * @param {number} last_frame last frame 
 18  * @param {number} animation_speed_factor value of animation speed factor
 19  */
 20 function activateAssetPointsAnimations(
 21     geojson_asset_points,
 22     geojson_assetpoints_directions,
 23     geojson_anchors,
 24     geojson_antennas,
 25     wall_intersections_points,
 26     geojson_lines_of_sight,
 27     geojson_walls,
 28     pulsing_dots_layers_sources,
 29     asset_point_animations,
 30     cpu_frame_rate,
 31     last_frame,
 32     animation_speed_factor
 33 ) {
 34     animation_speed_factor = parseFloat(document.getElementById("animation_speed_slider").value);
 35 
 36     geojson_asset_points.features.forEach(index => {
 37         var currentAssetFeatureId = index.properties.id;
 38         if (!animatingAssetPointFlags[currentAssetFeatureId]) {
 39             // call Animation
 40             animateAssetPointWithCustomMove(
 41                 index,
 42                 geojson_asset_points,
 43                 geojson_assetpoints_directions,
 44                 geojson_anchors,
 45                 geojson_antennas,
 46                 wall_intersections_points,
 47                 geojson_lines_of_sight,
 48                 geojson_walls,
 49                 pulsing_dots_layers_sources,
 50                 asset_point_animations,
 51                 cpu_frame_rate,
 52                 last_frame,
 53                 animation_speed_factor
 54             );
 55         }
 56     });
 57 
 58     // set printable Message
 59     printableMessage = "Animation speed factor has been set to '" + animation_speed_factor + "x'.";
 60 
 61     // print to console and to console_debugger
 62     console.log(printableMessage);
 63     printOnConsoleDebugger(printableMessage);
 64 }

 67 /**
 68  * Stop Asset Point Animations
 69  * @param {FeatureCollection} geojson_asset_points Asset Points FeatureCollection
 70  */
 71 function stopAssetPointsAnimations(geojson_asset_points) {
 72     geojson_asset_points.features.forEach(index => {
 73         var currentAssetFeatureId = index.properties.id;
 74 
 75         if (animatingAssetPointFlags[currentAssetFeatureId]) {
 76             // set animatingPointFlags to false
 77             animatingAssetPointFlags[currentAssetFeatureId] = false;
 78 
 79             // remove 'trace' layer and source if already exists on map
 80             if (map.getLayer(`trace-ap${currentAssetFeatureId + 1}`)) {
 81                 map.removeLayer(`trace-ap${currentAssetFeatureId + 1}`);
 82             }
 83             if (map.getSource(`trace-ap${currentAssetFeatureId + 1}`)) {
 84                 map.removeSource(`trace-ap${currentAssetFeatureId + 1}`);
 85             }
 86 
 87             // set printable Message
 88             printableMessage3 = "Stop of Asset Point Animations... ";
 89 
 90             // print to console and to console_debugger
 91             console.log(printableMessage3);
 92             printOnConsoleDebugger(printableMessage3);
 93         }
 94     });
 95 }

 98 /**
 99  * Async function to animate Asset Point with a custom move
100  * @async
101  * @param {number} index Asset Point index ID
102  * @param {FeatureCollection} geojson_asset_points Asset Points FeatureCollection
103  * @param {FeatureCollection} geojson_assetpoints_directions Asset Points Directions FeatureCollection
104  * @param {FeatureCollection} geojson_anchors Anchors FeatureCollection
105  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection
106  * @param {FeatureCollection} wall_intersections_points Map Features Walls Points Intersections FeatureCollection
107  * @param {FeatureCollection} geojson_lines_of_sight Lines of Sight FeatureCollection
108  * @param {FeatureCollection} geojson_walls Map Features Walls FeatureCollection
109  * @param {Array} pulsing_dots_layers_sources Pulsing Dots sources Array
110  * @param {Array} animations Pulsing Dots sources Array
111  * @param {number} cpu_frame_rate CPU frame rate
112  * @param {number} last_frame last frame 
113  * @param {number} animation_speed_factor value of animation speed factor
114  * @returns {null} null if no animations is setted
115  */
116 async function animateAssetPointWithCustomMove(
117     index,
118     geojson_asset_points,
119     geojson_assetpoints_directions,
120     geojson_anchors,
121     geojson_antennas,
122     wall_intersections_points,
123     geojson_lines_of_sight,
124     geojson_walls,
125     pulsing_dots_layers_sources,
126     animations,
127     cpu_frame_rate,
128     last_frame,
129     animation_speed_factor
130 ) {
131 
132     // Find the index of the first symbol layer in the map style
133     const layers = map.getStyle().layers;
134     let firstLinesOfSightLayerId;
135     for (const layer of layers) {
136         if (layer.id === 'layer-with-lines') {
137             firstLinesOfSightLayerId = layer.id;
138             break;
139         }
140     }
141 
142     var animation_geojson = '';
143     var color = '';
144     var currentAssetFeatureId = index.properties.id;
145 
146     if (!animatingAssetPointFlags[currentAssetFeatureId]) {
147         switch (index.properties.animation_option_index) {
148             case 1:
149                 printableMessage = `Animation 1 was selected for AP ${currentAssetFeatureId + 1}.`
150                 animation_geojson = animations[0]
151                 color = '#69d4ff';
152                 break;
153             case 2:
154                 printableMessage = `Animation 2 was selected for AP ${currentAssetFeatureId + 1}.`
155                 animation_geojson = animations[1]
156                 color = '#ad96ff';
157                 break;
158             case 3:
159                 printableMessage = `Animation 3 was selected for AP ${currentAssetFeatureId + 1}.`
160                 animation_geojson = animations[2]
161                 color = '#96ffb1';
162                 break;
163             case 4:
164                 printableMessage = `Animation 4 was selected for AP ${currentAssetFeatureId + 1}.`
165                 animation_geojson = animations[3]
166                 color = '#ff944d';
167                 break;
168             case 5:
169                 printableMessage = `Animation 5 was selected for AP ${currentAssetFeatureId + 1}.`
170                 animation_geojson = animations[4]
171                 color = '#fcd303';
172                 break;
173             case 6:
174                 printableMessage = `Animation 6 was selected for AP ${currentAssetFeatureId + 1}.`
175                 animation_geojson = animations[5]
176                 color = '#6bfc03';
177                 break;
178             case 7:
179                 printableMessage = `Animation 7 was selected for AP ${currentAssetFeatureId + 1}.`
180                 animation_geojson = animations[6]
181                 color = '#0341fc';
182                 break;
183             case 8:
184                 printableMessage = `Animation 8 was selected for AP ${currentAssetFeatureId + 1}.`
185                 animation_geojson = animations[7]
186                 color = '#de0202';
187                 break;
188             case 9:
189                 printableMessage = `Animation 9 was selected for AP ${currentAssetFeatureId + 1}.`
190                 animation_geojson = animations[8]
191                 color = '#57ffe6';
192                 break;
193             case 10:
194                 printableMessage = `Animation 10 was selected for AP ${currentAssetFeatureId + 1}.`
195                 animation_geojson = animations[9]
196                 color = '#90efe6';
197                 break;
198             case 11:
199                 printableMessage = `Animation 11 was selected for AP ${currentAssetFeatureId + 1}.`
200                 animation_geojson = animations[10]
201                 color = '#9050e6';
202                 break;
203             default:
204                 printableMessage = `Default animation null was selected for AP ${currentAssetFeatureId + 1}.`
205                 animation_geojson = null
206                 break;
207         }
208         console.log(printableMessage);
209         printOnConsoleDebugger(printableMessage)
210 
211         if (animation_geojson == null) {
212             printableMessage = `Animation for Asset Point ${currentAssetFeatureId + 1} was not setted!`
213             console.log(printableMessage);
214             printOnConsoleDebugger(printableMessage)
215             return;
216         }
217 
218         // We fetch the JSON here so that we can parse and use it separately
219         // from GL JS's use in the added source.
220         const response = await fetch(animation_geojson);
221         const data = await response.json();
222 
223         // save full coordinate list for later
224         const coordinates = data.features[0].geometry.coordinates;
225 
226         // set animatingPointFlag to true
227         animatingAssetPointFlags[currentAssetFeatureId] = true;
228 
229         // set disable 'Start Animation' button and enable 'Stop Animation' button
230         document.getElementById('btn_start_animations').disabled = true;
231         document.getElementById('btn_stop_animations').disabled = false;
232         document.getElementById('draw-assetpoint').disabled = true;
233         document.getElementById('delete-assetpoint').disabled = true;
234         document.getElementById('draw-antenna').disabled = true;
235         document.getElementById('delete-antenna').disabled = true;
236         document.getElementById('draw-anchor').disabled = true;
237         document.getElementById('delete-anchor').disabled = true;
238 
239         // update button color
240         updateAnimationColorButtonsOnStart();
241 
242         // start by showing just the first coordinate
243         data.features[0].geometry.coordinates = [coordinates[0]];
244 
245         // remove 'trace' layer and source if already exists on map
246         if (map.getLayer(`trace-ap${currentAssetFeatureId + 1}`)) {
247             map.removeLayer(`trace-ap${currentAssetFeatureId + 1}`);
248         }
249         if (map.getSource(`trace-ap${currentAssetFeatureId + 1}`)) {
250             map.removeSource(`trace-ap${currentAssetFeatureId + 1}`);
251         }
252 
253         // set 1st printableMessage 
254         printableMessage1 = `Animate Asset Point ${currentAssetFeatureId + 1}... `;
255 
256         // print to console and to console_debugger
257         console.log(printableMessage1);
258         printOnConsoleDebugger(printableMessage1);
259 
260         // add 'trace' layer and source
261         map.addSource(`trace-ap${currentAssetFeatureId + 1}`, { type: 'geojson', data: data });
262         map.addLayer({
263             'id': `trace-ap${currentAssetFeatureId + 1}`,
264             'type': 'line',
265             'source': `trace-ap${currentAssetFeatureId + 1}`,
266             'paint': {
267                 'line-color': color,
268                 'line-opacity': 0.5,
269                 'line-width': 6,
270             }
271         }, firstLinesOfSightLayerId);
272 
273         // setup the viewport on coordinate index 0
274         map.flyTo({ 'center': coordinates[0], 'zoom': 21, 'speed': 0.5 });
275 
276         // on a regular basis, add more coordinates from the saved list and update the map
277         let i = 0;
278         const timer = setInterval(async () => {
279             // get current date and time
280             const currentDate = new Date();
281             var now = currentDate.getTime();
282 
283             // limit to (cpu_frame_rate/10) updates per second
284             if ((now - last_frame) < cpu_frame_rate) {
285                 return;
286             };
287 
288             // Call event to close all open popups
289             map.fire('closeAllPopups');
290 
291             if (i < coordinates.length && animatingAssetPointFlags[currentAssetFeatureId]) {
292                 // save last position coordinate
293                 lastCoord = geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates;
294                 // save new position coordinate
295                 newCoord = coordinates[i];
296 
297                 // get bearing between two locations
298                 directionLargeLobe = getBearingBetweenLocations(lastCoord, newCoord);
299 
300                 // set large and small lobes of asset point
301                 geojson_asset_points.features[currentAssetFeatureId].properties.angle_direction_large_lobe = directionLargeLobe;
302                 geojson_asset_points.features[currentAssetFeatureId].properties.angle_direction_small_lobe = rescrictTo180Degrees(parseFloat(directionLargeLobe) + 180);
303 
304                 // push actual animation coordinate
305                 data.features[0].geometry.coordinates.push(coordinates[i]);
306                 geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates = coordinates[i];
307 
308                 // Call setData to the source layer `trace` on it.
309                 map.getSource(`trace-ap${currentAssetFeatureId + 1}`).setData(data)
310                 // Call setData to the source layer `geojson_asset_points` on it.
311                 map.getSource('asset-points').setData(geojson_asset_points);
312 
313                 // remove (splice) the direction at currentAssetFeatureId from 'geojson_assetpoints_directions' FeatureCollection
314                 geojson_assetpoints_directions.features.splice(currentAssetFeatureId * 2, 2);
315 
316                 // create assetpoint Lobe Direction at currentAssetFeatureId
317                 createAssetPointLobesDirection(geojson_asset_points.features[currentAssetFeatureId], geojson_assetpoints_directions)
318 
319                 // Call setData to the source layer `places` on it.
320                 map.getSource('assetpoints_directions').setData(geojson_assetpoints_directions);
321 
322                 // Print the coordinates of where the point had
323                 // finished being moved to on the map.
324                 messageToPrint =
325                     `<strong>${geojson_asset_points.features[currentAssetFeatureId].properties.title} Coordinates: </strong><br /> ` +
326                     `&#8226; Longitude:  ${rescrictTo180Degrees(geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates[0])} ` +
327                     `(${convertDMSLng(rescrictTo180Degrees(geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates[0]))}) <br /> ` +
328                     `&#8226; Latitude:  ${geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates[1]} ` +
329                     `(${convertDMSLat(geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates[1])})`;
330                 // Update container innerHTML
331                 updateContainerInnerHtml(coordinates_point_container, messageToPrint);
332 
333                 // set 2nd printableMessage 
334                 printableMessage2 = `<strong>Coordinates (Long, Lat):</strong><br />` +
335                     `(${geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates[0]}, ` +
336                     `${geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates[1]})`;
337 
338                 // print to console and to console_debugger
339                 console.log(printableMessage2);
340                 printOnConsoleDebugger(printableMessage2);
341 
342                 // Start the LinesOfSight animation.
343                 animateLinesOfSight(geojson_asset_points, geojson_anchors, geojson_antennas, geojson_lines_of_sight);
344                 // Start the IntersectionPoints animation.
345                 animateWallIntersectionPoints(wall_intersections_points, geojson_lines_of_sight, geojson_walls);
346                 // Start the pulsingDotPoints animation.
347                 animatePulsingDotPoints(pulsing_dots_layers_sources);
348                 // update last_fame with now time
349                 last_frame = now;
350                 // increment index
351                 i++;
352 
353             } else {
354                 window.clearInterval(timer);
355 
356                 // set animatingAssetPointFlag to false
357                 animatingAssetPointFlags[currentAssetFeatureId] = false;
358 
359                 // set enable 'Start Animation' button and disable 'Stop Animation' button
360                 document.getElementById('btn_start_animations').disabled = false;
361                 document.getElementById('btn_stop_animations').disabled = true;
362                 document.getElementById('draw-assetpoint').disabled = false;
363                 document.getElementById('delete-assetpoint').disabled = false;
364                 document.getElementById('draw-antenna').disabled = false;
365                 document.getElementById('delete-antenna').disabled = false;
366                 document.getElementById('draw-anchor').disabled = false;
367                 document.getElementById('delete-anchor').disabled = false;
368 
369                 // update button color
370                 updateAnimationColorButtonsOnStop();
371 
372                 // set 3rd printableMessage
373                 printableMessage3 = `End of Asset Point ${currentAssetFeatureId + 1} Animation!`;
374 
375                 // print to console and to console_debugger
376                 console.log(printableMessage3);
377                 printOnConsoleDebugger(printableMessage3);
378             }
379         }, 1000 / animation_speed_factor);
380     }
381 }
