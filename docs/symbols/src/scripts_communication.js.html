 84 /**
 85  * Add Antennas Data To Json Message
 86  * @param {string} payloadString Payload string data to add
 87  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection
 88  * @param {number} geojson_precison_decimal_places Value of precison decimal places
 89  * @returns {string} Payload string data
 90  */
 91 function addAntennasDataToJsonMessage(payloadString, geojson_antennas, geojson_precison_decimal_places) {
 92     // add 'antennas' to the JSON message
 93     payloadString += '"ant": [';
 94 
 95     // iterate by all others 'geojson_antennas' features
 96     for (let index = 0, len = geojson_antennas.features.length; index < len; index++) {
 97         // save long, lat, direction and opening values by index of antennas
 98         var antennaLong = geojson_antennas.features[index].geometry.coordinates[0].toFixed(geojson_precison_decimal_places)
 99         var antennaLat = geojson_antennas.features[index].geometry.coordinates[1].toFixed(geojson_precison_decimal_places)
100         var antennaDir = geojson_antennas.features[index].properties.angle_direction.toFixed(2);
101         var antennaOp = geojson_antennas.features[index].properties.angle_opening.toFixed(2);
102         var antennaTxPower = geojson_antennas.features[index].properties.tx_power;
103 
104         // Add the antennas positions until last
105         if (index < len - 1) {
106             payloadString += `{"LongLat": [${antennaLong + "," + antennaLat}], "DirOpen": [${antennaDir + "," + antennaOp}], "TxPower": ${antennaTxPower}}, `;
107         }
108         // Add the last antenna position 
109         if (index == len - 1) {
110             payloadString += `{"LongLat": [${antennaLong + "," + antennaLat}], "DirOpen": [${antennaDir + "," + antennaOp}], "TxPower": ${antennaTxPower}}`;
111         }
112     }
113     payloadString += ']';
114     return payloadString;
115 }

222 /**
223  * Construct Json Message to send to MQTT Broker
224  * @param {string} origin Label with description from where the message came from 
225  * @param {FeatureCollection} geojson_asset_points Asset Points FeatureCollection
226  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection
227  * @param {FeatureCollection} geojson_anchors Anchors FeatureCollection
228  * @param {number} geojson_precison_decimal_places Value of precison decimal places
229  * @param {number} backend_read_rate Backend read rate value
230  * @param {number} selectedMapIndex Index of selected Map
231  * @param {string} messageToSend MQTT message to send to broker
232  * @returns {string} Constructed Message to send to the MQTT Broker
233  */
234 function constructJsonMessage(
235     origin = "default",
236     geojson_asset_points,
237     geojson_antennas,
238     geojson_anchors,
239     geojson_precison_decimal_places,
240     backend_read_rate,
241     selectedMapIndex,
242     messageToSend
243 ) {
244     if (geojson_asset_points.features.length > 0 && geojson_antennas.features.length > 0) {
245         // var to create the message string - Open JSON structure
246         var payloadString = '{';
247 
248         // addSessionUuidToJsonMessage
249         payloadString = addSessionUuidToJsonMessage(payloadString);
250         payloadString += ', '
251 
252         // Full message (default)
253         if (origin == "default") {
254             // addStatusOpenToJsonMessage
255             payloadString = addStatusOpenToJsonMessage(payloadString);
256             payloadString += ', '
257 
258             // addSessionUuidToJsonMessage
259             payloadString = addMapIdToJsonMessage(payloadString, selectedMapIndex);
260             payloadString += ', '
261 
262             // addAntennasDataToJsonMessage
263             payloadString = addAntennasDataToJsonMessage(payloadString, geojson_antennas, geojson_precison_decimal_places);
264             payloadString += ', '
265             // addAnchorsDataToJsonMessage
266             if (geojson_anchors.features.length > 0) {
267                 payloadString = addAnchorsDataToJsonMessage(payloadString, geojson_anchors, geojson_precison_decimal_places);
268                 payloadString += ', '
269             }
270             // addAssetPointsDataToJsonMessage
271             payloadString = addAssetPointsDataToJsonMessage(payloadString, geojson_asset_points, geojson_precison_decimal_places);
272             payloadString += ', '
273             // addSelectedMlAgentsDataToJsonMessage
274             payloadString = addSelectedMlAgentsDataToJsonMessage(payloadString);
275             payloadString += ', '
276             // addRssiParametersDataToJsonMessage
277             payloadString = addRssiParametersDataToJsonMessage(payloadString);
278             payloadString += ', '
279             // addBackendReadRateToJsonMessage
280             payloadString = addBackendReadRateToJsonMessage(payloadString, backend_read_rate);
281         }
282 
283         // Partial message to update asset point values
284         else if (origin == "assetPoint") {
285             // addStatusOpenToJsonMessage
286             payloadString = addStatusOpenToJsonMessage(payloadString);
287             payloadString += ', '
288             // addAssetPointsDataToJsonMessage
289             payloadString = addAssetPointsDataToJsonMessage(payloadString, geojson_asset_points, geojson_precison_decimal_places);
290         }
291 
292         // Partial message to update antenna values
293         else if (origin == "antennas") {
294             // addStatusOpenToJsonMessage
295             payloadString = addStatusOpenToJsonMessage(payloadString);
296             payloadString += ', '
297             // addAntennasDataToJsonMessage
298             payloadString = addAntennasDataToJsonMessage(payloadString, geojson_antennas, geojson_precison_decimal_places);
299         }
300 
301         // Partial message to update anchors values
302         else if (origin == "anchors") {
303             // addStatusOpenToJsonMessage
304             payloadString = addStatusOpenToJsonMessage(payloadString);
305             payloadString += ', '
306             // addAntennasDataToJsonMessage
307             payloadString = addAnchorsDataToJsonMessage(payloadString, geojson_anchors, geojson_precison_decimal_places);
308         }
309 
310         // Partial message to update Rssi Parameters values
311         else if (origin == "rssiUpdateParams") {
312             // addStatusOpenToJsonMessage
313             payloadString = addStatusOpenToJsonMessage(payloadString);
314             payloadString += ', '
315             // addRssiParametersDataToJsonMessage
316             payloadString = addRssiParametersDataToJsonMessage(payloadString);
317         }
318 
319         // Partial message to update ML Agents algorithms
320         else if (origin == "algs") {
321             // addStatusOpenToJsonMessage
322             payloadString = addStatusOpenToJsonMessage(payloadString);
323             payloadString += ', '
324             // addSelectedMlAgentsDataToJsonMessage
325             payloadString = addSelectedMlAgentsDataToJsonMessage(payloadString);
326         }
327 
328         // Send a message to broker to inform all subscribers about DISCONNECT
329         else if (origin == "disconnect") {
330             // addStatusCloseToJsonMessage
331             payloadString = addStatusCloseToJsonMessage(payloadString);
332         }
333 
334         // close JSON structure
335         payloadString += '}';
336         // populate messageToSend with payloadString 
337         messageToSend = payloadString
338     }
339     return messageToSend;
340 }

342 /**
343  * Process Json Message received from MQTT Broker
344  * @param {string} message MQTT message received from the broker
345  * @param {FeatureCollection} geojson_asset_points Asset Points FeatureCollection
346  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection
347  * @param {FeatureCollection} geojson_anchors Anchors FeatureCollection
348  * @param {Array} pulsing_dots_layers_sources Pulsing Dots sources Array
349  * @param {string} received_uuid MQTT received session UUID
350  */
351 function processReceivedJsonMessage(message, geojson_asset_points, geojson_anchors, geojson_antennas, pulsing_dots_layers_sources, received_uuid) {
352     // first replace (') by (")
353     var result = message.payloadString.replace(/'/g, '"');
354     // process received JSON message format 
355     jsonData = JSON.parse(result);
356     // save received_uuid
357     received_uuid = jsonData["uuid"];
358 
359     // verify if received_uuid is session_uuid 
360     if (received_uuid == session_uuid) {
361         if (jsonData["from"] == "backend") {
362             // deal with calculations to display info on container
363             // save received data from JSON
364             ap_distance_values = jsonData["ap-dist"]
365             ap_activ_values = jsonData["ap-activ"];
366             ap_rssi_values = jsonData["ap-rssi"];
367             ap_wall_intersections = jsonData["ap-wall-inter"];
368             ant_dir_ap_intersections = jsonData["ant-dir-ap-inter"];
369             anchors_dists = jsonData["anch-dist"];
370             anchors_activ_values = jsonData["anch-activ"];
371             anchors_rssi_values = jsonData["anch-rssi"];
372             anchors_wall_intersections = jsonData["anch-wall-inter"];
373             ant_dir_anchors_intersections = jsonData["ant-dir-anch-inter"];
374 
375             // set message and populate to print on console and on Distances & RSSI container
376             var messageToContainer = `<strong>Distances [m] | RSSI [dBm] | Walls: # | Directionated: (boolean)</strong></br>`;
377 
378             // check length of received data
379             if (geojson_asset_points.features.length == ap_distance_values.length) {
380                 // iterate over geojson_asset_points
381                 geojson_asset_points.features.forEach(index_asset_point => {
382                     // populate message to print
383                     messageToContainer += `AP ${index_asset_point.properties.id + 1}:</br>`
384 
385                     // iterate over geojson_antennas
386                     geojson_antennas.features.forEach(index_antenna => {
387                         // restrict the length of distance_value and rssi_value to 2 decimal points.
388                         var distance_value = Math.round(parseFloat(Object.values(ap_distance_values)[index_asset_point.properties.id][index_antenna.properties.id]) * 100) / 100
389                         var rssi_value = Math.round(parseFloat(Object.values(ap_rssi_values)[index_asset_point.properties.id][index_antenna.properties.id]) * 100) / 100
390 
391                         // save antenna name by index
392                         var antenna_name = geojson_antennas.features[index_antenna.properties.id].properties.title;
393                         // save wall intersections by line
394                         var wall_intersections_by_line = ap_wall_intersections[index_asset_point.properties.id][index_antenna.properties.id];
395                         // save direction intersections by line
396                         var dir_intersections_by_line = ant_dir_ap_intersections[index_asset_point.properties.id][index_antenna.properties.id]
397                         // populate message to print
398                         messageToContainer += ` &#8226; ${antenna_name}: ${distance_value} m | RSSI: ${rssi_value} dBm | Walls: ${wall_intersections_by_line} | Dir: ${String(dir_intersections_by_line)}</br>`;
399                     });
400                 });
401             }
402             else {
403                 messageToContainer += ' • Updating Asset Points data ... </br>';
404             }
405             // add anchors data if added
406             if (anchors_dists.length > 0) {
407                 if (anchors_dists.length == geojson_anchors.features.length) {
408                     // populate message to print
409                     messageToContainer += `Anchors: </br>`
410 
411                     // iterate over geojson_anchors
412                     geojson_anchors.features.forEach(index_anchor => {
413                         // save anchor name by index
414                         var anchor_name = `Anchor ${index_anchor.properties.id + 1}`;
415                         // populate message to print
416                         messageToContainer += ` • ${anchor_name}: `;
417 
418                         // iterate over geojson_antennas 
419                         geojson_antennas.features.forEach(index_antenna => {
420                             // restrict the length of anchor_dist to 2 decimal points.
421                             var anchor_dist_value = Math.round(parseFloat(anchors_dists[index_anchor.properties.id][index_antenna.properties.id]) * 100) / 100
422 
423                             // Add until the last antenna anchor distance
424                             if (index_antenna.properties.id < geojson_antennas.features.length - 1) {
425                                 // populate message to print
426                                 messageToContainer += `${anchor_dist_value} m | `;
427                             }
428                             // Add the last antenna anchor distance 
429                             if (index_antenna.properties.id == geojson_antennas.features.length - 1) {
430                                 messageToContainer += `${anchor_dist_value} m </br>`;
431                             }
432                         });
433                     });
434                 }
435                 else {
436                     messageToContainer += 'Updating Anchors data ... </br>';
437                 }
438             }
439             // Update container innerHTML
440             updateContainerInnerHtml(distance_rssi_container, messageToContainer)
441         }
442         else if (jsonData["from"] == "predictor") {
443             var pulsing_dots_layers_IDs_enabled = []
444             var pulsing_dots_layers_IDs_disabled = []
445             
446             // verify possibility of receiving all ML algorithms
447             if (selected_ml_agent_algorithm == '"all"') {
448                 selected_ml_agent_algorithm = ["knn", "svr", "gbr", "rf", "dt", "fp_dt", "fp_rf", "fp_rnn"]
449             }
450 
451             var ordered_algorithms_dict = { 'KNN': 0, 'SVR': 1, "GBR": 2, "RF": 3, "DT": 4, "FP_DT": 5, "FP_RF": 6, "FP_RNN": 7 }
452 
453             // for the firsts pulsing dots = 8 
454             const pulsing_dots_fixed_size = pulsing_dots_layers_sources[0].features.length;
455 
456             // check if received data has the same size as geojson_asset_points
457             if (jsonData["coords"].length == geojson_asset_points.features.length) {
458                 // iterate over 'predictor' message
459                 for (let i = 0, len_i = jsonData["coords"].length; i < len_i; i++) {
460                     // save ap_number = ap index + 1
461                     const ap_number = geojson_asset_points.features[i].properties.id + 1;
462                     // Deal with coordinates to pulsing dots
463                     // iterate over 'pulsing_dots_algorithms' array
464                     for (let a = 0, len = pulsing_dots_fixed_size; a < len; a++) {
465                         // save algorithm name to create layerID
466                         const algo_name = pulsing_dots_layers_sources[i].features[a].properties.algorithm.replace(/"/g, '').toUpperCase();
467                         // create the layerID
468                         const layerID = `layer-with-pulsing-dot-ap${ap_number}-${algo_name}`;
469                         // verify if message has values for each AP
470                         if (jsonData["coords"][i].length > 0) {
471                             // flag to check if algorithm already exists
472                             var enabled = false;
473                             // iterate over 'coords data' array
474                             for (let j = 0, len_j = jsonData["coords"][i].length; j < len_j; j++) {
475                                 // check if jsonData "coords" array have some 'pulsing_dots_algorithms'
476                                 if (jsonData["coords"][i][j][2] == `${algo_name}`) {
477                                     if (geojson_asset_points.features[i].properties.show_predictions) {
478                                         // calculate pulsing dot ID index
479                                         const pulsing_dot_ID = ordered_algorithms_dict[algo_name];
480 
481                                         // save coordinates of current index j
482                                         long = parseFloat(jsonData["coords"][i][j][0]);
483                                         lat = parseFloat(jsonData["coords
