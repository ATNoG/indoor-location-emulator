1551         const currentDate = new Date();

1554         // Limit to (cpu_frame_rate/10) updates per second
1555         if ((now - last_frame) < cpu_frame_rate) {
1556             return;
1557         };
1558 
1559         // save coordinates from event
1560         const coords = e.lngLat;
1561         // Set a UI indicator for dragging.
1562         canvas.style.cursor = 'grabbing';
1563         // Call event to close all open popups
1564         map.fire('closeAllPopups');
1565 
1566         // save last position coordinate
1567         lastCoord = geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates;
1568         // save new position coordinate
1569         newCoord = [rescrictTo180Degrees(coords.lng), coords.lat];
1570         // get bearing between two locations
1571         directionLargeLobe = getBearingBetweenLocations(lastCoord, newCoord);
1572         // set large and small lobes of asset point
1573         geojson_asset_points.features[currentAssetFeatureId].properties.angle_direction_large_lobe = directionLargeLobe;
1574         geojson_asset_points.features[currentAssetFeatureId].properties.angle_direction_small_lobe = rescrictTo180Degrees(parseFloat(directionLargeLobe) + 180);
1575         // Update the point feature coordinates in `geojson_asset_points` 
1576         geojson_asset_points.features[currentAssetFeatureId].geometry.coordinates = [rescrictTo180Degrees(coords.lng), coords.lat];
1577         // And call setData to the source layer `point` on it.
1578         map.getSource('asset-points').setData(geojson_asset_points);
1579 
1580         // remove (splice) the direction at currentAssetFeatureId from 'geojson_assetpoints_directions' FeatureCollection (x2 for asset_points_directions)
1581         geojson_assetpoints_directions.features.splice(currentAssetFeatureId * 2, 2);
1582         // create assetpoint Lobe Direction at currentAssetFeatureId
1583         createAssetPointLobesDirection(geojson_asset_points.features[currentAssetFeatureId], geojson_assetpoints_directions)
1584         // Call setData to the source layer `places` on it.
