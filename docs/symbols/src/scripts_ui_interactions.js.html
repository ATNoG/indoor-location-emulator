115 /**

119  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection

125  * @param {string} messageToSend MQTT message to send to broker

128 function updateSelectedMlAgents(
129     connectedFlag,
130     geojson_asset_points,
131     geojson_antennas,
132     geojson_anchors,
133     geojson_precison_decimal_places,
134     backend_read_rate,

142

145         selected_ml_agent_algorithm.push(`"${index.id.toLowerCase()}"`)
146     });
147 
148     if (selected_ml_agent_algorithm.length < 1) {
149         printableMessage = "Please select a ML-agent to be used in positioning predictions.";
150 
151         // update buttons colors
152         updateUnselectedColorButtonSubmitAgents();
153         updateUnselectedColorButtonStartConnection();
154     }
155 
156     else {
157         if (selected_ml_agent_algorithm.length > 1) {
158             if (selected_ml_agent_algorithm == '"knn","svr","gbr","rf","dt","fp_dt","fp_rf","fp_rnn"') {
159                 selected_ml_agent_algorithm = '"all"';
160             }
161 
162             printableMessage = "The selected ML-agents to be used in positioning predictions are '" + selected_ml_agent_algorithm + "'.";
163         }
164         else if (selected_ml_agent_algorithm.length == 1) {
165             printableMessage = "The selected ML-agent to be used in positioning predictions is '" + selected_ml_agent_algorithm + "'.";
166         }
167 
168         // update buttons colors
169         updateSelectedColorButtonSubmitAgents();
170         updateSelectedColorButtonStartConnection();
171     }
172 
173     // print to console and to console_debugger
174     console.log(printableMessage);
175     printOnConsoleDebugger(printableMessage);
176 
177     // Construct Json Message
178     if (connectedFlag)
179         if (selected_ml_agent_algorithm.length >= 1) {
180             messageToSend = constructJsonMessage(
181                 "algs",
182                 geojson_asset_points,
183                 geojson_antennas,
184                 geojson_anchors,
185                 geojson_precison_decimal_places,
186                 backend_read_rate,
187                 selectedMapIndex,
188                 messageToSend
189             );
190             publishMqttMessage(mqttTopicToPublish, messageToSend);
191         }
192         else {
193             return prev;
194         }
195 
196     return selected_ml_agent_algorithm;
197 }

366 /**
367  * Generate an Animated Image with rgb basecolor
368  * @param {string} basecolor RGB color values
369  * @returns Animated Pulsing Dot
370  */
371 function generateAnimatedImage(basecolor) {
372     // ========== Create Animated Pulsing Dot ==========
373     // This implements `StyleImageInterface`
374     // to draw a pulsing dot icon on the map.
375     const size = 256; // The image will be 256 pixels square.
376     const bytesPerPixel = 4; // Each pixel is represented by 4 bytes: red, green, blue, and alpha.
377     const animatedPulsingDot = {
378         width: size,
379         height: size,
380         data: new Uint8Array(size * size * bytesPerPixel),
381 
382         onAdd: function () {
383             const canvas = document.createElement('canvas');
384             canvas.width = this.width;
385             canvas.height = this.height;
386             this.context = canvas.getContext('2d');
387             this.map = map;
388         },
389 
390         render: function () {
391             const duration = 1000;
392             const t = (performance.now() % duration) / duration;
393 
394             const radius = (size / 2) * 0.3;
395             const outerRadius = (size / 2) * 0.7 * t + radius;
396             const context = this.context;
397 
398             // Draw the outer circle.
399             context.clearRect(0, 0, this.width, this.height);
400             context.beginPath();
401             context.arc(
402                 this.width / 2,
403                 this.height / 2,
404                 outerRadius,
405                 0,
406                 Math.PI * 2
407             );
408             context.fillStyle = `rgba(${basecolor}, ${1 - t}})`;
409             context.fill();
410 
411             // Draw the inner circle.
412             context.beginPath();
413             context.arc(
414                 this.width / 2,
415                 this.height / 2,
416                 radius,
417                 0,
418                 Math.PI * 2
419             );
420             context.fillStyle = `rgba(${basecolor}, 0.5)`;
421             context.strokeStyle = 'white';
422             context.lineWidth = 2 + 4 * (1 - t);
423             context.fill();
424             context.stroke();
425 
426             // Update this image's data with data from the canvas.
427             this.data = context.getImageData(
428                 0,
429                 0,
430                 this.width,
431                 this.height
432             ).data;
433 
434             // Continuously repaint the map, resulting
435             // in the smooth animation of the dot.
436             this.map.triggerRepaint();
437 
438             // Return `true` to let the map know that the image was updated.
439             return true;
440         }
441     }
442     return animatedPulsingDot;
443 }

445 /**
446  * Auto scroll down element (Console debug content)
447  * @param {HTMLElement} htmlElement HTML Element (Console debug)
448  * @param {string} string Text to print on console
449  */
450 function autoScrollDown(htmlElement, string) {
451     // Allow 1px inaccuracy by adding 1
452     var isScrolledToBottom = htmlElement.scrollHeight - htmlElement.clientHeight <= htmlElement.scrollTop + 1;
453     var newElement = document.createElement("div");
454     newElement.innerHTML = "> " + string + "</br>";
455     htmlElement.appendChild(newElement);
456     // scroll to bottom if isScrolledToBottom
457     if (isScrolledToBottom)
458         htmlElement.scrollTop = htmlElement.scrollHeight - htmlElement.clientHeight;
459 }

463  * @param {string} string Text to print on console
464  */
465 function printOnConsoleDebugger(string) {
466     var element = document.getElementById('console_debugger');
467     // auto scroll down
468     autoScrollDown(element, string);

606 /**
607  * Update Asset Point Configs option inside popup
608  * @param {string} source_obj Source of FeatureCollection 
609  * @param {FeatureCollection} geojson_assetpoints Asset Points FeatureCollection 
610  * @param {FeatureCollection} geojson_assetpoints_directions Asset Points Directions FeatureCollection 
611  * @param {number} prev_large_lobe_angle_direction Previous value of large lobe angle direction of Asset Point
612  * @param {number} prev_small_lobe_angle_direction Previous value of small lobe angle direction of Asset Point
613  * @param {number} animation_item Asset Point Animation item index
614  * @param {number} currentFeatureId Asset Point feature index
615  */
616 function updateAssetPointConfigs(source_obj, geojson_assetpoints, geojson_assetpoints_directions, prev_large_lobe_angle_direction, prev_small_lobe_angle_direction, animation_item, currentFeatureId) {
617     // Update the Point feature in `places` coordinates
618     var currentObj = geojson_assetpoints.features.find(obj => {
619         return obj.properties.id === currentFeatureId;
620     })
621 
622     // save parameters
623     var large_lobe_angle_direction = rescrictTo180Degrees(document.getElementById("ap_large_direction_parameter").value).toFixed(2);
624     var small_lobe_angle_direction = rescrictTo180Degrees(document.getElementById("ap_small_direction_parameter").value).toFixed(2);
625     var large_lobe_angle_opening = document.getElementById("ap_large_opening_parameter").value;
626     var small_lobe_angle_opening = document.getElementById("ap_small_opening_parameter").value;
627 
628     // parse values
629     if (prev_large_lobe_angle_direction != String(large_lobe_angle_direction)) {
630         large_angle_direction_value = parseFloat(large_lobe_angle_direction);
631         small_angle_direction_value = parseFloat(rescrictTo180Degrees(parseFloat(large_lobe_angle_direction) + 180));
632     }
633     else if (prev_small_lobe_angle_direction != String(small_lobe_angle_direction)) {
634         large_angle_direction_value = parseFloat(rescrictTo180Degrees(parseFloat(small_lobe_angle_direction) + 180));
635         small_angle_direction_value = parseFloat(small_lobe_angle_direction);
636     }
637     else {
638         large_angle_direction_value = parseFloat(large_lobe_angle_direction);
639         small_angle_direction_value = parseFloat(small_lobe_angle_direction);
640     }
641 
642     large_angle_opening_value = parseFloat(large_lobe_angle_opening);
643     small_angle_opening_value = parseFloat(small_lobe_angle_opening);
644 
645     // set values
646     geojson_assetpoints.features[currentFeatureId].properties.angle_direction_large_lobe = large_angle_direction_value;

728 /**
729  * Create Asset Point Lobes Direction with provided a rotation angles
730  * @param {number} index Asset point feature index
731  * @param {FeatureCollection} geojson_assetpoints_directions Asset Point Directions FeatureCollection
732  * @returns {FeatureCollection} Asset Points Directions FeatureCollection
733  */
734 function createAssetPointLobesDirection(index, geojson_assetpoints_directions) {
735     // first save antenna coordinates
736     var assetpoint_coordinates = index.geometry.coordinates;
737     // value of rotation of large lobe in decimal degrees, anti-clockwise
738     var angle_direction_large = -index.properties.angle_direction_large_lobe;
739     // value of rotation of small lobe in decimal degrees, anti-clockwise
740     var angle_direction_small = -index.properties.angle_direction_small_lobe;
741 
742     // use turf.js - Sector
743     // Creates a circular sector of a circle of given radius and center Point, between (clockwise) bearing1 and bearing2; 
744     // 0 bearing is North of center point, positive clockwise.
745     var center = turf.point([assetpoint_coordinates[0], assetpoint_coordinates[1]]);
746     var radiusFactor = 1000;
747     var radius1 = 1 / radiusFactor * 0.75;
748     var radius2 = 1 / radiusFactor * 0.55;
749 
750     var bearingLarge1 = 0 - index.properties.angle_opening_large_lobe / 2;
751     var bearingLarge2 = bearingLarge1 + index.properties.angle_opening_large_lobe;
752 
753     var bearingSmall1 = 0 - index.properties.angle_opening_small_lobe / 2;;
754     var bearingSmall2 = bearingSmall1 + index.properties.angle_opening_small_lobe;
755 
756     var sectorLobeLarge = turf.sector(center, radius1, bearingLarge1, bearingLarge2);
757     var sectorLobeSmall = turf.sector(center, radius2, bearingSmall1, bearingSmall2);
758 
759     // apply rotation
760     var options = { pivot: [assetpoint_coordinates[0], assetpoint_coordinates[1]] };
761     var rotatedSectorLobeLarge = turf.transformRotate(sectorLobeLarge, angle_direction_large, options);
762     var rotatedSectorLobeSmall = turf.transformRotate(sectorLobeSmall, angle_direction_small, options);
763 
764     // push (splice) the 'sector' object in 'index.properties.id', deleting 0 elements (x2 for asset_points_directions)
765     geojson_assetpoints_directions.features.splice(index.properties.id * 2, 0, rotatedSectorLobeLarge, rotatedSectorLobeSmall);
766 
767     // copy properties (x2 for asset_points_directions)
768     geojson_assetpoints_directions.features[index.properties.id * 2].properties = index.properties;
769     geojson_assetpoints_directions.features[index.properties.id * 2 + 1].properties = index.properties;
770 
771     return geojson_assetpoints_directions;
772 }

774 /**
775  * Create Antenna Direction with provided a rotation angle
776  * @param {number} index Antenna feature index
777  * @param {*} geojson_antennas_directions Antennas Directions FeatureCollection
778  * @returns {FeatureCollection} Antennas Directions FeatureCollection
779  */
780 function createAntennaDirection(index, geojson_antennas_directions) {
781     // first save antenna coordinates
782     var antenna_coordinates = index.geometry.coordinates;
783 
784     // value of rotation in decimal degrees, anti-clockwise
785     var angle_direction = -index.properties.angle_direction;
786 
787     // use turf.js - Sector
788     // Creates a circular sector of a circle of given radius and center Point, between (clockwise) bearing1 and bearing2; 
789     // 0 bearing is North of center point, positive clockwise.
790     var center = turf.point([antenna_coordinates[0], antenna_coordinates[1]]);
791     var radiusFactor = 1000;
792     var radius = 1 / radiusFactor;
793 
794     var bearing1 = 0 - index.properties.angle_opening / 2;
795     var bearing2 = bearing1 + index.properties.angle_opening;
796     var sector = turf.sector(center, radius, bearing1, bearing2);
797 
798     // apply rotation
799     var options = { pivot: [antenna_coordinates[0], antenna_coordinates[1]] };
800     var rotatedSector = turf.transformRotate(sector, angle_direction, options);
801 
802     // push (splice) the 'sector' object in 'index.properties.id', deleting 0 elements
803     geojson_antennas_directions.features.splice(index.properties.id, 0, rotatedSector)
804 
805     // copy antennas properties
806     geojson_antennas_directions.features[index.properties.id].properties = index.properties;
807 
808     return geojson_antennas_directions;
809 }

813  * @param {FeatureCollection} geojson_objs Points (Asset Points or Anchors) FeatureCollection data
814  * @param {FeatureCollection} geojson_antennas Antennas FeatureCollection
815  * @returns {Array} Lines Of Sight Array data
816  */
817 function catchLinesOfSight(geojson_objs, geojson_antennas) {
818     // set empty array data
819     var lines = [];
820 
821     // Iterate by all the 'geojson_antennas.features', using array.forEach(index => { // do something here });
822     geojson_antennas.features.forEach(antenna_index => {
823         // Iterate by all the 'geojson_antennas.features', using array.forEach(index => { // do something here });

825             if (objs_index.properties.show_lines) {
